(function(){
  var initial = "int i = 5;\n" +

    "sortableId": "sortable",
    "max_wrong_lines": 10,
    "grader": ParsonsWidget._graders.LineBasedGrader,
    "exec_limit": 2500,
    "can_indent": true,

  });

  parsonsPuzzle.shuffleLines();

      event.preventDefault(); 
      parsonsPuzzle.shuffleLines();

      event.preventDefault();

  });
